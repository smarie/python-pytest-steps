36 class ExceptionHook(object):
37 """
38 A context manager used to register a hook for exceptions.
39 This hook (a method provided in constructor) will be called if an exception is caught.
40 The exception will then be raised as usual.
41  
42 Example:
43 --------
44 >>>with ExceptionHook(print):
45 >>> assert False
46  
47 The hook method ('print' in the above example) will be called with the type of exception, the exception value,
48 and the exception traceback, before the exception is actually raised.
49 """
50 def __init__(self, exc_handler):
51 """
52 Constructor.
53  
54 :param exc_handler: the method that should be called if an exception is raised inside this context manager.
55 """
56 self.exc_handler = exc_handler
57  
58 def __enter__(self):
59 pass
60  
61 def __exit__(self, exc_type, exc_val, exc_tb):
62 if exc_type is not None:
63 self.exc_handler(exc_type, exc_val, exc_tb)
64  
65 # return False so that the exception is always raised
66 return False

159 def one_fixture_per_step(*args):
160 """
161 A decorator for a function-scoped fixture so that it works well with generator-mode test functions. You do not have
162 to use it in parametrizer mode, although it does not hurt.
163  
164 By default if you do not use this decorator but use the fixture in a generator-mode test function, only the
165 fixture created for the first step will be injected in your test function, and all subsequent steps will see that
166 same instance.
167  
168 Decorating your fixture with `@one_fixture_per_step` tells `@test_steps` to transparently replace the fixture object
169 instance by the one created for each step, before each step executes in your test function. This results in all
170 steps using different fixture instances, as expected.
171  
172 It is recommended that you put this decorator as the second decorator, right after `@pytest.fixture`:
173  
174 ```python
175 @pytest.fixture
176 @one_fixture_per_step
177 def my_cool_fixture():
178 return random()
179 ```
180  
181 Note: When a fixture is decorated with `@one_fixture_per_step`, the object that is injected in your test function
182 is a transparent proxy of the fixture, so it behaves exactly like the fixture. If for some reason you want to get
183 the "true" inner wrapped object, you can do so using `get_underlying_fixture(my_fixture)`.
184 :return:
185 """
186 if len(args) == 1 and callable(args[0]):
187 return one_fixture_per_step_decorate(args[0])
188 elif len(args) == 0:
189 return one_fixture_per_step_decorate
190 else:
191 raise ValueError("@one_fixture_per_step accepts no argument")

246 def __init__(self, step_names, test_function, first_step_args, first_step_kwargs):
247 """
248 Constructor with declaration of all step names in advance,
249 as well as the test function to execute and the first step args and kwargs
250  
251 Nothing will be executed here, the test function will only be called once to create the generator.
252  
253 :param step_names:
254 :param test_function:
255 :param first_step_args:
256 :param first_step_kwargs:
257 """
258 self.steps = step_names
259 self.exceptions = dict()
260  
261 # Remember objects that should be replaced in subsequent steps
262 # -- for positional arguments, store in a dict under key=position
263 self.replaceable_args = dict()
264 for i, a in enumerate(first_step_args):
265 if is_replacable_fixture_wrapper(a):
266 self.replaceable_args[i] = a
267  
268 # -- for keyword arguments, store in a dict under key=name
269 self.replaceable_kwargs = dict()
270 for k, a in first_step_kwargs.items():
271 if is_replacable_fixture_wrapper(a):
272 self.replaceable_kwargs[k] = a

274 # create the generator
275 self.gen = test_function(*first_step_args, **first_step_kwargs)

278 """
279 As of today a step can execute if there are no registered exceptions (in previous mandatory steps).
280 :param step_name:
281 :return:
282 """

285 def execute(self, step_name, args, kwargs):
286 """
287 Executes one iteration of the monitored generator.
288  
289 :param step_name:
290 :return:
291 """
292  
293 if self.can_execute(step_name):
294 # Replace all objects that should be replaced
295 for i, a in self.replaceable_args.items():
296 replace_fixture(a, args[i])
297 for k, a in self.replaceable_kwargs.items():
298 replace_fixture(a, kwargs[k])
299  
300 # Execute the step
301 with self._monitor(step_name):
302 try:
303 res = next(self.gen)
304 except StopIteration:
305 raise StepExecutionError(step_name)
306  
307 # Manage exceptions in optional steps
308 if res is None:
309 # We now accept None yields
310 # raise StepYieldError(step_name, None)
311 pass
312  
313 elif isinstance(res, string_types):
314 if res != step_name:
315 raise StepYieldError(step_name, res)
316  
317 elif isinstance(res, optional_step):
318 # optional step: check if the execution went well
319 if res.exec_result is None:
320 raise ValueError("Internal error: this should not happen")
321  
322 elif isinstance(res.exec_result, OptionalStepException):
323 # An exception happened in the optional step. We can now raise it safely
324 # (raising it sooner would break the generator)
325 reraise(res.exec_result.exc_type, res.exec_result.exc_val, res.exec_result.tb)
326  
327 elif isinstance(res.exec_result, _DependentTestsNotRunException):
328 # This exception has been put here to declare that the optional step did not run because a
329 # dependency is missing. >> Skip or fail
330 # TODO add fail_instead_of_skip argument like in depends_on
331 should_fail = False
332 msg = "This test step '%s' depends on other steps, and the following have failed: %s" \
333 "" % (step_name, res.exec_result.dependency_names)
334 if should_fail:
335 pytest.fail(msg)
336 else:
337 pytest.skip(msg)
338 elif res.exec_result is True:
339 # normal execution success
340 pass
341 else:
342 raise ValueError("Internal error: this should not happen")
343 else:
344 # the generator yielded a res that is not of an accepted type
345 raise StepYieldError(step_name, res)
346 else:
347 # A mandatory step failed before this one. The generator is broken, no need to even try >> Skip or fail
348 # TODO add fail_instead_of_skip argument like in depends_on
349 should_fail2 = False
350 failed_step = next(iter(self.exceptions.keys())) if len(self.exceptions) == 1 \
351 else list(self.exceptions.keys())
352 msg = "This test step '%s' is not run because non-optional previous step '%s' has failed" \
353 "" % (step_name, failed_step)
354 if should_fail2:
355 pytest.fail(msg)
356 else:
357 pytest.skip(msg)

533 class optional_step(object):
534 """
535 A context manager that you can use in *generator* mode in order to declare a step as independent, so that next
536 steps in the generator can continue to execute even if this one fails.
537  
538 When this context manager is used you should not forget to yield the context object ! Otherwise the test step will
539 be marked as successful even if it was not.
540  
541 You can optionally declare dependencies using the `depends_on=` argument in the constructor. If so, you should use
542 the .should_execute() method if you wish your code block to be properly skipped.
543  
544 ```python
545 from pytest_steps import test_steps, optional_step
546  
547 @test_steps('step_a', 'step_b', 'step_c', 'step_d')
548 def test_suite_opt():
549 # Step A
550 assert not False
551 yield
552  
553 # Step B
554 with optional_step('step_b') as step_b:
555 assert False
556 yield step_b
557  
558 # Step C depends on step B
559 with optional_step('step_c', depends_on=step_b) as step_c:
560 if step_c.should_run():
561 assert True
562 yield step_c
563  
564 # Step D
565 assert not False
566 yield
567 ```
568 """
569  
570 def __init__(self,
571 step_name, # type: str
572 depends_on=None # type: Union[optional_step, Iterable[optional_step]]
573 ):
574 """
575 Creates the context manager for an optional step named `step_name` with optional dependencies on other
576 optional steps.
577  
578 :param step_name: the name of this optional step. This name will be used in pytest failure/skip messages when
579 other steps depend on this one and are skipped/failed because this one was skipped/failed.
580 :param depends_on: an optional dependency or list of dependencies, that should all be optional steps created
581 with an `optional_step` context manager.
582 """
583 # default values
584 self.step_name = step_name
585 self.exec_result = None
586 self.depends_on = depends_on or []
587  
588 # coerce depends_on to a list
589 if not isinstance(self.depends_on, It):
590 self.depends_on = [self.depends_on]
591  
592 # dependencies should be optional steps too
593 for dependency in self.depends_on:
594 if not isinstance(dependency, optional_step):
595 raise ValueError("depends_on should only contain optional_step instances")
596  
597 def __str__(self):
598 return self.step_name
599  
600 def __enter__(self):
601 # check that all dependencies have run
602 for dependency in self.depends_on:
603 if not dependency.ran_with_success():
604 if self.exec_result is None:
605 self.exec_result = _DependentTestsNotRunException(self.step_name, dependency.step_name)
606 else:
607 self.exec_result.dependency_names.append(dependency.step_name)
608  
609 # Unfortunately if we raise an exception here it will not be caught by the __exit__ method
610 # So there is absolutely no way to prevent the code block to execute,
611 # - even with an ExitStack (I tried !): indeed the ExitStack is nothing more than a context manager so if an
612 # error is raised during its __enter__ method, then its __exit__ stack will not be called
613 # - A PEP 377 was created for that but was rejected
614 # - A hack also exists but it interferes with the debugger so it is too complex
615 # See https://stackoverflow.com/questions/12594148/skipping-execution-of-with-block
616  
617 return self

619 def should_run(self):
620 """
621 Return True if there are no exceptions, false otherwise
622 :return:
623 """
624 return self.exec_result is None

626 def __exit__(self, exc_type, exc_val, traceback):
627 # Store this step's execution result for later
628 if exc_type is None:
629 if self.exec_result is None:
630 # Success !
631 self.exec_result = True
632 else:
633 # there was a dependency that had a failure, we can not forget it
634 pass
635 else:
636 # Failure: remember the exception
637 self.exec_result = OptionalStepException(exc_type, exc_val, traceback)
638  
639 # We have to *cancel* the exception in the stack of the test function since it is a generator and we need to
640 # be able to execute subsequent steps
641 # see https://docs.python.org/3/reference/datamodel.html#object.__exit__
642 return True
